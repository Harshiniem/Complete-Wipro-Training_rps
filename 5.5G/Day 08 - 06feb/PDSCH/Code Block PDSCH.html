// ============================================
// 5G NR Code Block Concatenation
// Physical Downlink Shared Channel (PDSCH)
// ============================================

#include <iostream>
#include <vector>
#include <chrono>
using namespace std;

/**
 * @brief Concatenate multiple code blocks into a single stream
 * @param blocks Vector of code blocks (each block is a vector of bits)
 * @return Final concatenated bit stream
 * @complexity O(n) where n = total number of bits
 */
vector<int> concatenateCodeBlocks(const vector<vector<int>>& blocks) {
    vector<int> finalStream;
    
    // Pre-allocate memory for efficiency
    size_t totalSize = 0;
    for(const auto& block : blocks) {
        totalSize += block.size();
    }
    finalStream.reserve(totalSize);
    
    // Concatenate each block sequentially
    for(const auto& block : blocks) {
        finalStream.insert(finalStream.end(), block.begin(), block.end());
    }
    
    return finalStream;
}

int main() {
    // Example: Code blocks from Rate Matching stage
    vector<vector<int>> codeBlocks = {
        {1,0,1,1,0,1},  // Block 1
        {0,1,1,0,1,0},  // Block 2
        {1,1,0,0,1,1}   // Block 3
    };
    
    // Measure execution time
    auto start = chrono::high_resolution_clock::now();
    
    // Perform concatenation
    vector<int> finalBits = concatenateCodeBlocks(codeBlocks);
    
    auto end = chrono::high_resolution_clock::now();
    auto duration = chrono::duration_cast<chrono::microseconds>(end - start);
