Step 1: Polar Encoder Output

Step 2: Circular Buffer Concept

All 512 bits are stored in a circular buffer. This allows efficient bit selection for both shortening/puncturing (E < N) and repetition (E > N) cases.

Circular Buffer: 512 positions Seamless wrap-around

Step 4: Shortening / Puncturing

When E < N, we need to discard (N - E) bits. This is called puncturing. Bits are selected from the circular buffer, and some bits are not transmitted.

// 5G NR PDCCH Polar Rate Matching Implementation
#include <iostream>
#include <vector>
using namespace std;

// Rate Matching Function
vector<int> rateMatching(const vector<int>& input, int N, int E) {
    vector<int> output(E);
    
    // Circular buffer index
    int k = 0;
    
    for (int i = 0; i < E; i++) {
        // Read from circular buffer with wrap-around
        output[i] = input[k % N];
        k++;
    }
    
    return output;
}

// Main function
int main() {
    // Parameters
    int N = 512;  // Polar coded bits
    int AL = 4;   // Aggregation Level
    int E = AL * 108;  // Output bits = 432
    
    // Generate random input bits (Polar encoder output)
    vector<int> polarBits(N);
    for (int i = 0; i < N; i++) {
        polarBits[i] = rand() % 2;
    }
    
    // Perform Rate Matching
    vector<int> rateMatchedBits = rateMatching(polarBits, N, E);
    
    // Display results
    cout << "Input Bits (N): " << N << endl;
    cout << "Output Bits (E): " << E << endl;
    cout << "Aggregation Level: " << AL << endl;
    
    if (E > N) {
        cout << "Mode: REPETITION (" << (E - N) << " bits repeated)" << endl;
    } else if (E < N) {
        cout << "Mode: PUNCTURING (" << (N - E) << " bits discarded)" << endl;
    } else {
        cout << "Mode: NO RATE MATCHING" << endl;
    }
    
    // Print first 20 output bits
    cout << "Output (first 20 bits): ";
    for (int i = 0; i < min(20, E); i++) {
        cout << rateMatchedBits[i] << " ";
    }
    cout << "..." << endl;
    
    return 0;
}
